C++ 使用示例

本文档介绍如何使用 C++ 语言调用 HumanoidSDK 库，并演示如何构建一个简单的示例程序。

提示

更多示例请参考 sdk_package/example 目录。

配置环境变量

cd ~/sdk_package
export LD_LIBRARY_PATH="$(pwd)/lib/linux/x86_64/x64:$(pwd)/dependency/sdk/Runtime_Env/linux/x86_64/x64:$(pwd)/dependency/fastdds/linux/x86_64/x64:$LD_LIBRARY_PATH"

可选：如果要使用自定义 DDS 配置文件，设置 ROBOT_DDS_PROFILE 环境变量：

export ROBOT_DDS_PROFILE=path/to/your/Robot_DDS_Profile.xml

构建一个简单示例程序

1. 创建 main.cpp 文件

 1 // main.cpp
 2 #include <iostream>
 3 #include <iomanip>
 4 #include "h10-w/h10w_params.h"
 5 #include "common/common_def.h"
 6
 7 using namespace humanoid;
 8 using namespace humanoid::h10w;
 9
10 int main() {
11 H10wParams params;
12 JointLimit jointLimits;
13
14 if (params.getJointSoftLimit(jointLimits) == 0) {
15     std::cout << "get joint soft limit success" << std::endl;
16
17     for (const auto& limit : jointLimits) {
18     const auto& [jointIndex, maxPos, minPos] = limit;
19     std::cout << "    joint " << jointIndex << ": "
20                 << "max pos=" << std::fixed << std::setprecision(3) << maxPos
21                 << " rad, "
22                 << "min pos=" << minPos << " rad" << std::endl;
23     }
24 } else {
25     std::cout << "get joint soft limit failed" << std::endl;
26     return -1;
27 }
28 return 0;
29 }

2. 创建 CMakeLists.txt 文件

 1cmake_minimum_required(VERSION 3.10)
 2project(simple_example)
 3
 4# Project Options
 5set(CMAKE_CXX_STANDARD 17)
 6set(CMAKE_CXX_EXTENSIONS OFF)
 7set(CMAKE_CXX_STANDARD_REQUIRED ON)
 8
 9# Find HumanoidSDK
10set(HumanoidSDK_DIR "path/to/sdk_package/cmake/HumanoidSDK") ## replace with your sdk package path
11find_package(HumanoidSDK REQUIRED)
12
13add_executable(${PROJECT_NAME} main.cpp)
14target_link_libraries(${PROJECT_NAME} humanoid_sdk::humanoid_sdk)

3. 编译并运行示例程序

mkdir build && cd build && cmake .. && make
./simple_example
